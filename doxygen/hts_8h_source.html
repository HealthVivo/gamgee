275 static inline int hts_reg2bin(int64_t beg, int64_t end, int min_shift, int n_lvls)
276 {
277  int l, s = min_shift, t = ((1<<((n_lvls<<1) + n_lvls)) - 1) / 7;
278  for (--end, l = n_lvls; l > 0; --l, s += 3, t -= 1<<((l<<1)+l))
279  if (beg>>s == end>>s) return t + (beg>>s);
280  return 0;
281 }
