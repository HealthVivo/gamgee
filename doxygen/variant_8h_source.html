200  //Try to avoid calling bcf_update_alleles as it causes significant overhead in

204  if(m_body->rlen >= ref_length)
205  {
206  memcpy(m_body->d.allele[0], ref, ref_length); //memcpy is slightly faster than strcpy
207  m_body->d.allele[0][ref_length] = '\0'; //append null - why not include in memcpy? see set_reference_allele with char option
208  m_body->d.shared_dirty |= BCF1_DIRTY_ALS;
209  }
210  else
211  {
212  //expensive - causes significant reallocs within htslib

215  m_body->d.allele[0] = const_cast<char*>(ref);
216  //Re-use same d.allele char** as update_alleles writes to different region of memory anyway
217  bcf_update_alleles(const_cast<const bcf_hdr_t*>(m_header.m_header.get()), m_body.get(), const_cast<const char**>(m_body->d.allele), m_body->n_allele);
